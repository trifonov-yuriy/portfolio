О проекте

В этом проекте я выступил в роли чистого программиста — разработал полностью программную реализацию программатора для микроконтроллеров EFM8 (Silicon Labs) на базе ATmega328p. Аппаратная часть уже существовала, моя задача была — оживить её и реализовать протокол C2 "с нуля".

Результат: Работающий программатор, который через UART принимает HEX-файлы с ПК и прошивает EFM8, полностью заменяя дорогой оригинальный программатор Silicon Labs.

1. Реализация протокола C2 на регистрах

Самая сложная часть проекта — протокол C2. В отличие от I2C или SPI, его нет в железе AVR, поэтому пришлось эмулировать всё программно с точным соблюдением таймингов. 

// define.h - назначение пинов и макросы
#define C2_D                    PD2
#define C2_CLK                  PD3

// Макросы для переключения направления пина
#define C2_D_IN                 DDRD &= ~(1 << PD2)  // На вход
#define C2_D_OUT                DDRD |= (1 << PD2)   // На выход
#define C2_CLK_OUT              DDRD |= (1 << PD3)   // CLK всегда выход

// Установка уровней
#define C2_D_ON                 PORTD |= (1 << PD2)
#define C2_CLK_ON               PORTD |= (1 << PD3)
#define C2_D_OFF                PORTD &= ~(1 << PD2)
#define C2_CLK_OFF              PORTD &= ~(1 << PD3)

// Чтение входа
#define C2_D_READ               (PIND & (1 << PD2))
Прямая работа с регистрами DDRx и PORTx для максимальной производительности 
// c2.c - формирование тактового импульса с запретом прерываний
void c2_clockPulse() 
{
    cli();  // Запрещаем прерывания для точной временной диаграммы

    C2_CLK_OFF;     // 0 на C2CK
    _delay_us(2);   // 2 мкс точно по даташиту

    C2_CLK_ON;      // 1 на C2CK

    sei();          // Разрешаем прерывания
}
cli()/sei() гарантируют, что никакое прерывание не собьёт тайминги

2. Отправка битов и байтов

// c2.c - отправка последовательности битов
void c2_sendBits(uint8_t data, uint8_t length) 
{
    for(uint8_t i = 0; i < length; i += 1) 
    {
        if(data >> i & 0x01) 
            C2_D_ON;        // Бит = 1
        else 
            C2_D_OFF;       // Бит = 0

        c2_clockPulse();    // Тактовый импульс
    }
}

void c2_sendByte(uint8_t byte) 
{
    c2_sendBits(byte, 8);   // Отправляем 8 бит
}

void c2_sendStopBit() 
{
    c2_sendBits(0x00, 1);   // Стоп-бит
}
Универсальная функция для отправки любого количества бит

3. Чтение данных (двунаправленная линия)

C2D — двунаправленная линия, поэтому перед чтением нужно переключить пин на вход. 

// c2.c - чтение битов
uint8_t c2_readBits(uint8_t length) 
{
    uint8_t mask = 0x01 << (length - 1);
    uint8_t data = 0;

    C2_D_IN;          // Переключаем пин на вход
    C2_D_PULL_DOWN;   // Включаем подтяжку

    for (uint8_t i = 0; i < length; i += 1) 
    {
        c2_clockPulse();   // Тактовый импульс

        data >>= 1;
        if (C2_D_READ)     // Читаем бит с пина
            data |= mask;
    }
    
    C2_D_OUT;         // Возвращаем пин в режим выхода
    return data;
}

uint8_t c2_readByte() 
{
    return c2_readBits(8);
}
Динамическое переключение направления пина внутри одной операции

// isr.c - обработчик прерывания UART
volatile uint8_t c2_message[300];

volatile uint8_t readFlag = 0;

ISR(USART_RX_vect) 
{
    // Читаем байт из аппаратного буфера UART

    // Сохраняем в кольцевой буфер
    c2_message[c2_messagePtr] = received_data;

    readFlag = 1;
}
Кольцевой буфер на 300 байт для приёма HEX-файлов 
// usart.c - инициализация UART
void usart_init() 
{
    uint32_t ubbr = (F_CPU_ / (16 * USART_BAUD_RATE_)) - 1;
    
    UBRR0L = ubbr & 0xff;
    UBRR0H = (ubbr & 0xff00) >> 8;

    // Включаем приём, передачу и прерывание по приёму
    UCSR0B |= (1 << TXEN0) | (1 << RXEN0) | (1 << RXCIE0);
    
    // 8 бит данных
    UCSR0C |= (1 << UCSZ01) | (1 << UCSZ00);
}

7. Автоматическая калибровка 4 ESC

Полностью автоматический процесс калибровки всех ESC одновременно. 

// c2.c - обработка команды калибровки
case Calibration:
{
    if(!calibration_flag) 
    {
        calibration_flag = 1;    
        
        // Максимальный газ (100%)
        uint8_t throttle = ((100 * 63) / 100) + 62;
        
        // Устанавливаем газ для всех 4 моторов
        throttle_1 = throttle_2 = throttle_3 = throttle_4 = throttle;
        OCR0A = throttle;  // Таймер0 - мотор 1
        OCR1A = throttle;  // Таймер1A - мотор 2
        OCR1B = throttle;  // Таймер1B - мотор 3
        OCR2A = throttle;  // Таймер2 - мотор 4

        POWER_ON;           // Включаем питание
        _delay_ms(5000);    // Ждём 5 секунд (ESC запоминают максимум)

        // Минимальный газ (0%)
        throttle = ((0 * 63) / 100) + 62;
        
        throttle_1 = throttle_2 = throttle_3 = throttle_4 = throttle;
        OCR0A = OCR1A = OCR1B = OCR2A = throttle;
        
        _delay_ms(5000);    // Ждём завершения калибровки
        POWER_OFF;
    }
} break;
Одной командой калибруются сразу 4 ESC — экономия времени на производстве

if(timer == 0)

// define.h - все макросы для работы с железом

// Управление мультиплексором выбора мотора
#define A_PIN   PD4
#define B_PIN   PD5

#define ESC_1  {PORTD &= ~(1 << A_PIN); PORTD &= ~(1 << B_PIN);}  // A=0 B=0
#define ESC_2  {PORTD |=  (1 << A_PIN); PORTD &= ~(1 << B_PIN);}  // A=1 B=0
#define ESC_3  {PORTD &= ~(1 << A_PIN); PORTD |=  (1 << B_PIN);}  // A=0 B=1
#define ESC_4  {PORTD |=  (1 << A_PIN); PORTD |=  (1 << B_PIN);}  // A=1 B=1

// Управление питанием
#define POWER_PIN           PC1
#define POWER_ON            (PORTC |= (1 << POWER_PIN))
#define POWER_OFF           (PORTC &= ~(1 << POWER_PIN))

// Управление проверкой питания
#define POWER_CHECK_PIN     PD7
#define POWER_CHECK_ON      (PORTD |= (1 << POWER_CHECK_PIN))
#define POWER_CHECK_OFF     (PORTD &= ~(1 << POWER_CHECK_PIN))

// Выбор канала ADC
#define POWER_CHECK_ADC     (ADMUX = (ADMUX & 0xf8) | 3)  // Канал 3
